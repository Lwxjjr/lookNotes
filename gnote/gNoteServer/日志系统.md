# 表结构
```go
// log_stash/model.go
type LogModel struct {
	ID          uint      `json:"id" gorm:"primaryKey"`
	CreatedAt   time.Time `json:"createdAt" gorm:"column:createdAt"`
	UpdatedAt   time.Time `json:"updatedAt" gorm:"column:updatedAt"`
	IP          string    `json:"ip"`
	Addr        string    `json:"addr"`
	Level       Level     `json:"level"`
	Title       string    `json:"title"`                       // 标题
	Content     string    `json:"content"`                     // 内容
	UserID      uint      `json:"userID" gorm:"column:userID"` // 用户ID
	UserName    string    `json:"userName" gorm:"column:userName"`
	ServiceName string    `json:"serviceName" gorm:"column:serviceName"`
	Status      bool      `json:"status"`                                             // 登录状态
	Type        LogType   `json:"type"`                                               // 日志类型 1 登录 2 操作 3 运行
	ReadStatus  bool      `json:"readStatus" gorm:"column:readStatus; default:false"` // 阅读状态 true 已读 false 未读
}
```
